cmake_minimum_required (VERSION 3.0.0)
project (SimpleMatchmaker)
add_subdirectory (enet)
file(GLOB_RECURSE source_files "src/*.cpp")
add_executable(SimpleMatchmaker ${source_files})
target_link_libraries(SimpleMatchmaker enet)
target_include_directories(SimpleMatchmaker PRIVATE "enet/include/")
if(WIN32)
  target_link_libraries(SimpleMatchmaker Winmm ws2_32)
endif()
